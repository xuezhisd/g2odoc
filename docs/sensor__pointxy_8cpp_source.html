41  bool SensorPointXY::isVisible(SensorPointXY::WorldObjectType* to){
42  if (! _robotPoseObject)
43  return false;
44 
45  assert(to && to->vertex());
46  VertexType* v=to->vertex();
47  VertexType::EstimateType pose=v->estimate();
48  VertexType::EstimateType delta = _robotPoseObject->vertex()->estimate().inverse()*pose;
49  Vector2d translation=delta;
50  double range2=translation.squaredNorm();
51  if (range2>_maxRange2)
52  return false;
53  if (range2<_minRange2)
54  return false;
55  translation.normalize();
56  double bearing=acos(translation.x());
57  if (fabs(bearing)>_fov)
58  return false;
59  return true;
60  }
