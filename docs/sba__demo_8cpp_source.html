67 static double gauss_rand(double mean, double sigma)
68 {
69  double x, y, r2;
70  do {
71  x = -1.0 + 2.0 * uniform_rand(0.0, 1.0);
72  y = -1.0 + 2.0 * uniform_rand(0.0, 1.0);
73  r2 = x * x + y * y;
74  } while (r2 > 1.0 || r2 == 0.0);
75  return mean + sigma * y * std::sqrt(-2.0 * log(r2) / r2);
76 }
