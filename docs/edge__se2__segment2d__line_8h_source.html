52  {

54  const VertexSegment2D* l2 = static_cast<const VertexSegment2D*>(_vertices[1]);

56  Vector2D predP1 = iEst * l2->estimateP1();

61  predP1.dot(normal)*.5 + predP2.dot(normal)*.5);
