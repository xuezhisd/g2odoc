50 struct IncrementalEdgesCompare
51 {
52  bool operator()(const EdgeInformation& e1, const EdgeInformation& e2)
53  {
54  int i11 = e1.fromId, i12 = e1.toId;
55  if (i11 > i12)
56  swap(i11, i12);
57  int i21 = e2.fromId, i22 = e2.toId;
58  if (i21 > i22)
59  swap(i21, i22);
60  if (i12 < i22)
61  return true;
62  if (i12 > i22)
63  return false;
64  return i11 < i21;
65  }
66 };
